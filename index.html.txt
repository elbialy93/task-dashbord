<script>
const teams = [
  { id:'data',   label:'إدخال البيانات', stages:['استلام الملف','جاري الرفع','مراجعة','تعديل','اعتماد نهائي'] },
  { id:'dev',    label:'البرمجة',        stages:['مهمة مراجعة','جاري','ملاحظات','منتهي'] },
  { id:'design', label:'التصميم',        stages:['استلام السكريبت','تصميم أولي','مراجعة','تعديل','تسليم نهائي'] }
];
const priLabel = { high:'عالية', med:'متوسطة', low:'منخفضة' };
const priTag   = { high:'tag-red', med:'tag-amber', low:'tag-green' };

// البيانات الافتراضية في حال كانت الذاكرة فارغة
let defaultTasks = {
  data: [
    { id:1, name:'امتحانات الرياضيات - الفصل الأول', emp:'أحمد', stage:0, pri:'high' },
    { id:2, name:'امتحانات العلوم - المرحلة الإعدادية', emp:'سارة', stage:1, pri:'med' }
  ],
  dev: [],
  design: []
};

let tasks = {};
let nextId = Date.now(), activeTeam = 'data', currentTeam = 'data';

// دالة لاستعادة البيانات من ذاكرة المتصفح
function loadData() {
    const saved = localStorage.getItem('testat_dashboard_data');
    if (saved) {
        tasks = JSON.parse(saved);
    } else {
        tasks = defaultTasks;
    }
}

// دالة لحفظ البيانات في ذاكرة المتصفح
function saveData() {
    localStorage.setItem('testat_dashboard_data', JSON.stringify(tasks));
}

function totalTasks(){ return Object.values(tasks).flat().length; }
function doneTasks(){
  let d=0;
  teams.forEach(t=>{ tasks[t.id].forEach(tk=>{ if(tk.stage===t.stages.length-1) d++; }); });
  return d;
}
function inProgressTasks(){
  let p=0;
  teams.forEach(t=>{ tasks[t.id].forEach(tk=>{ if(tk.stage>0&&tk.stage<t.stages.length-1) p++; }); });
  return p;
}

function renderStats(){
  const total=totalTasks(), done=doneTasks(), prog=inProgressTasks();
  const pct=total?Math.round(done/total*100):0;
  document.getElementById('stats-grid').innerHTML=`
    <div class="stat"><div class="stat-label">إجمالي المهام</div><div class="stat-val">${total}</div></div>
    <div class="stat"><div class="stat-label">منتهية</div><div class="stat-val">${done}</div></div>
    <div class="stat"><div class="stat-label">جاري التنفيذ</div><div class="stat-val">${prog}</div></div>
    <div class="stat"><div class="stat-label">نسبة الإنجاز</div><div class="stat-val">${pct}%</div></div>
  `;
}

function renderTabs(){
  document.getElementById('tabs').innerHTML=teams.map(t=>`
    <div class="tab${t.id===activeTeam?' active':''}" onclick="switchTeam('${t.id}')">
      ${t.label} <span style="font-size:11px;opacity:0.6">(${tasks[t.id].length})</span>
    </div>`).join('');
}

function renderSections(){
  document.getElementById('sections').innerHTML=teams.map(t=>`
    <div class="section${t.id===activeTeam?' active':''}" id="sec-${t.id}">
      <button class="add-btn" onclick="openModal('${t.id}')">+ إضافة مهمة</button>
      <div class="kanban" id="kanban-${t.id}"></div>
    </div>`).join('');
  teams.forEach(t=>renderKanban(t));
}

function renderKanban(team){
  const el=document.getElementById('kanban-'+team.id);
  if(!el) return;
  el.innerHTML=team.stages.map((stage,si)=>{
    const cards=tasks[team.id].filter(tk=>tk.stage===si);
    return `<div class="col">
      <div class="col-head">${stage}<span class="col-count">${cards.length}</span></div>
      ${cards.length===0?'<div class="empty-col">لا توجد مهام</div>':''}
      ${cards.map(tk=>`
        <div class="card">
          <div class="card-title">${tk.name}</div>
          <div class="card-emp">${tk.emp}</div>
          <div class="card-tags">
            <span class="tag tag-blue">${tk.emp}</span>
            <span class="tag ${priTag[tk.pri]}">${priLabel[tk.pri]}</span>
          </div>
          <div class="card-actions">
            ${si>0?`<button class="act-btn" onclick="moveTask('${team.id}',${tk.id},-1)">← سابق</button>`:''}
            ${si<team.stages.length-1?`<button class="act-btn" onclick="moveTask('${team.id}',${tk.id},1)">التالي →</button>`:''}
            <button class="act-btn act-del" onclick="deleteTask('${team.id}',${tk.id})">حذف</button>
          </div>
        </div>`).join('')}
    </div>`;
  }).join('');
}

function switchTeam(id){
  activeTeam=id;
  renderTabs();
  document.querySelectorAll('.section').forEach(s=>s.classList.remove('active'));
  document.getElementById('sec-'+id).classList.add('active');
}

function openModal(teamId){
  currentTeam=teamId;
  ['f-name','f-emp','f-notes'].forEach(id=>document.getElementById(id).value='');
  document.getElementById('f-pri').value='med';
  document.getElementById('modal-title').textContent='إضافة مهمة — فريق '+teams.find(t=>t.id===teamId).label;
  document.getElementById('modal-bg').classList.add('open');
}

function closeModal(){ document.getElementById('modal-bg').classList.remove('open'); }

function saveTask(){
  const name=document.getElementById('f-name').value.trim();
  const emp=document.getElementById('f-emp').value.trim();
  if(!name||!emp) return;
  tasks[currentTeam].push({ id:Date.now(), name, emp, stage:0, pri:document.getElementById('f-pri').value });
  closeModal(); 
  saveData(); // حفظ البيانات
  refresh();
}

function moveTask(teamId,taskId,dir){
  const team=teams.find(t=>t.id===teamId);
  const tk=tasks[teamId].find(t=>t.id===taskId);
  const ns=tk.stage+dir;
  if(ns<0||ns>=team.stages.length) return;
  tk.stage=ns;
  saveData(); // حفظ البيانات
  refresh();
}

function deleteTask(teamId,taskId){
  tasks[teamId]=tasks[teamId].filter(t=>t.id!==taskId);
  saveData(); // حفظ البيانات
  refresh();
}

function refresh(){ 
    renderStats(); 
    renderTabs(); 
    renderSections(); 
}

document.getElementById('modal-bg').addEventListener('click',function(e){ if(e.target===this) closeModal(); });

// نقطة البداية
loadData();
refresh();
</script>
